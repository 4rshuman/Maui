<script>
const card = document.getElementById("card");
const music = document.getElementById("music");
const message = document.getElementById("message");

// Updated text for Baba
const text = "Dear Baba, Happy Birthday! I wanted to create something special to show how much I appreciate everything you do. You've always been my greatest mentor and my coolest friend. I am truly grateful for the wisdom you share and the many happy moments we've enjoyed together. You make everything look easy, and your strength is something I always look up to. Wishing you a day filled with relaxation, joy, and all the things you love. You deserve the very best! Love you always.";

let i = 0;
let started = false;

function typeWriter() {
  if (i < text.length) {
    message.innerHTML += text.charAt(i);
    i++;
    setTimeout(typeWriter, 45);
    message.scrollTop = message.scrollHeight; 
  }
}

card.onclick = () => {
  card.classList.toggle("open");
  music.play().catch(e => console.log("Audio interaction required"));

  if (!started) {
    typeWriter();
    goldFlakes();
    started = true;
  }
};

function goldFlakes() {
  const colors = ['#f1c40f', '#ffffff', '#e67e22'];
  for (let i = 0; i < 50; i++) {
    let p = document.createElement("div");
    p.className = "petal";
    p.style.left = Math.random() * 100 + "vw";
    p.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
    p.style.animationDuration = (3 + Math.random() * 4) + "s";
    p.style.width = (Math.random() * 8 + 5) + "px";
    document.body.appendChild(p);
  }
}
</script>
